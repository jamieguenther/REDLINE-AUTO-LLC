102  private List<SoldVehicle> getSalesReport(List<SoldVehicle> soldVehicles) {
103 
104  Map<String, Integer> result = new HashMap<>();
105  DecimalFormat df2 = new DecimalFormat("#.##");
106  ArrayList<String> makes = new ArrayList<String>();
107  ArrayList<String> keys = new ArrayList<String>();
108  HashMap<String, Integer> makesAndPrice = new HashMap<String, Integer>();
109  String mostSoldMake = "";
110  String leastSoldMake = "";
111  String mostMake = "";
112  String leastMake = "";
113  double totalNumberOfVehiclesSold = 0;
114 
115  for (int i = 0; i < soldVehicles.size(); i++) {
116  makes.add(soldVehicles.get(i).getMake());
117  makesAndPrice.put(soldVehicles.get(i).getMake(), soldVehicles.get(i).getPrice());
118  }
119 
120  for (String unique : new HashSet<>(makes)) {
121  result.put(unique, Collections.frequency(makes, unique));
122  }
123 
124  for (int numberOfVehicles : result.values()) {
125  totalNumberOfVehiclesSold += numberOfVehicles;
126  }
127 
128  double max = Collections.max(result.values());
129  double min = Collections.min(result.values());
130 
131  for (Entry<String, Integer> entry : result.entrySet()) {
132  if (entry.getValue() == max) {
133  keys.add(entry.getKey());
134  mostMake = entry.getKey();
135  }
136  if (entry.getValue() == min) {
137  keys.add(entry.getKey());
138  leastMake = entry.getKey();
139  }
140  }
141 
142  mostSoldMake = df2.format(((max / totalNumberOfVehiclesSold) * 100)) + "% of your vehicles sold are: "
143  + mostMake;
144  leastSoldMake = df2.format(((min / totalNumberOfVehiclesSold) * 100)) + "% of your vehicles sold are: "
145  + leastMake;
146 
147  for (SoldVehicle sv : soldVehicles) {
148  sv.setMostSoldVehicle(mostSoldMake);
149  sv.setLeastSoldVehicle(leastSoldMake);
150  }
151 
152  return soldVehicles;
